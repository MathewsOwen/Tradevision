document.addEventListener('DOMContentLoaded', () => { const graphArea = document.getElementById('graphArea'); for (let i = 0; i < 30; i++) { const candle = document.createElement('div'); candle.className = 'candle'; candle.style.height = `${Math.floor(Math.random() * 200) + 50}px`; graphArea.appendChild(candle); } setInterval(() => { document.querySelectorAll('.candle').forEach(c => { let h = parseInt(c.style.height); c.style.height = `${Math.max(20, h + (Math.random() * 20 - 10))}px`; }); }, 1000); });
